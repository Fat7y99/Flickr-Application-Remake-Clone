showAlertDialog(BuildContext context, String str) {
  // Create button
  Widget okButton = TextButton(
    child: Text("OK"),
    onPressed: () {
      if (str == 'Everything is ready') {
        Navigator.of(context).pop();
        Navigator.of(context).pop();
      } else
        Navigator.of(context).pop();
    },
  );

  AlertDialog alert = AlertDialog(
    title: Text("Alert"),
    content: Text(str),
    actions: [
      okButton,
    ],
  );

  // show the dialog
  showDialog(
    context: context,
    builder: (BuildContext context) {
      return alert;
    },
  );
}
